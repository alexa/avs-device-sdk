25 enum class NotificationsCapabilityAgentState {
26  // The capability agent is awaiting directives.
27  IDLE,
28  // A call to renderNotification() has been made, but the notification has not
29  // yet finished rendering.
30  PLAYING,
31  // A call to cancelNotificationRendering() has been made, but the notification
32  // has not yet been cancelled.
33  CANCELING_PLAY,
34  // The capability agent is shutting down.
35  SHUTTING_DOWN,
36  // The capability agent has finished shutting down.
37  SHUTDOWN
38 };

40 inline std::string stateToString(NotificationsCapabilityAgentState state) {
41  switch (state) {
42  case NotificationsCapabilityAgentState::IDLE:
43  return "IDLE";
44  case NotificationsCapabilityAgentState::PLAYING:
45  return "PLAYING";
46  case NotificationsCapabilityAgentState::CANCELING_PLAY:

50  case NotificationsCapabilityAgentState::SHUTDOWN:
