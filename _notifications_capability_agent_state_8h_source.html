23 enum class NotificationsCapabilityAgentState {
24  // The capability agent is awaiting directives.
25  IDLE,
26  // A call to renderNotification() has been made, but the notification has not yet finished rendering.
27  PLAYING,
28  // A call to cancelNotificationRendering() has been made, but the notification has not yet been cancelled.
29  CANCELING_PLAY,
30  // The capability agent is shutting down.
31  SHUTTING_DOWN,
32  // The capability agent has finished shutting down.
33  SHUTDOWN
34 };

36 inline std::string stateToString(NotificationsCapabilityAgentState state) {
37  switch (state) {

39  return "IDLE";
40  case NotificationsCapabilityAgentState::PLAYING:
41  return "PLAYING";
42  case NotificationsCapabilityAgentState::CANCELING_PLAY:

45  return "SHUTTING_DOWN";
46  case NotificationsCapabilityAgentState::SHUTDOWN:
