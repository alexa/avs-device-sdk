184 template <typename Task, typename... Args>
185 auto Executor::pushTo(bool front, Task task, Args&&... args) -> std::future<decltype(task(args...))> {
186  // Remove arguments from the tasks type by binding the arguments to the task.
187  auto boundTask = std::bind(std::forward<Task>(task), std::forward<Args>(args)...);
188 
189  /*
190  * Create a std::packaged_task with the correct return type. The decltype only returns the return value of the
191  * boundTask. The following parentheses make it a function call with the boundTask return type. The package task
192  * will then return a future of the correct type.
193  *
194  * Note: A std::packaged_task fulfills its future *during* the call to operator(). If the user of a
195  * std::packaged_task hands it off to another thread to execute, and then waits on the future, they will be able to
196  * retrieve the return value from the task and know that the task has executed, but they do not know exactly when
197  * the task object has been deleted. This distinction can be significant if the packaged task is holding onto
198  * resources that need to be freed (through a std::shared_ptr for example). If the user needs to wait for those
199  * resources to be freed they have no way of knowing how long to wait. The translated_task lambda below is a
200  * workaround for this limitation. It executes the packaged task, then disposes of it before passing the task's
201  * return value back to the future that the user is waiting on.
202  */
203  using PackagedTaskType = std::packaged_task<decltype(boundTask())()>;
204  auto packaged_task = std::make_shared<PackagedTaskType>(boundTask);
205 
206  // Create a promise/future that we will fulfill when we have cleaned up the task.
207  auto cleanupPromise = std::make_shared<std::promise<decltype(task(args...))>>();
208  auto cleanupFuture = cleanupPromise->get_future();
209 
210  // Remove the return type from the task by wrapping it in a lambda with no return value.
211  auto translated_task = [packaged_task, cleanupPromise]() mutable {
212  // Execute the task.
213  packaged_task->operator()();
214  // Note the future for the task's result.
215  auto taskFuture = packaged_task->get_future();
216  // Clean up the task.
217  packaged_task.reset();
218  // Forward the task's result to our cleanup promise/future.
219  forwardPromise(cleanupPromise, &taskFuture);
220  };
221 
222  // Release our local reference to packaged task so that the only remaining reference is inside the lambda.
223  packaged_task.reset();
224 
225  {
226  bool restart = false;
227  std::lock_guard<std::mutex> queueLock{m_queueMutex};
228  if (!m_shutdown) {
229  restart = !m_threadRunning;
230  m_queue.emplace(front ? m_queue.begin() : m_queue.end(), std::move(translated_task));
231  } else {
232  using FutureType = decltype(task(args...));
233  return std::future<FutureType>();
234  }
235 
236  if (restart) {
237  // Restart task thread.
238  m_taskThread.start(std::bind(&Executor::runNext, this));
239  m_threadRunning = true;
240  }
241  }
242 
243  m_delayedCondition.notify_one();
244  return cleanupFuture;
245 }
