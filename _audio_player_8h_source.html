95 inline std::string playerStateToString(AudioPlayerState state) {
96  switch (state) {
97  case AudioPlayerState::IDLE:
98  return "IDLE";
99  case AudioPlayerState::PLAYING:

101  case AudioPlayerState::STOPPED:
102  return "STOPPED";

104  return "PAUSED";

106  return "BUFFERING";

109  case AudioPlayerState::FINISHED:
110  return "FINISHED";
111  }
112  return "unknown AudioPlayerState";
113 }
