83 inline std::ostream& operator<<(std::ostream& stream, const MediaDescription& mediaDescription) {
84  switch (mediaDescription.mixingBehavior) {
85  case sdkInterfaces::audio::MixingBehavior::BEHAVIOR_PAUSE:
86  stream << "BEHAVIOR_PAUSE";
87  break;
88  case sdkInterfaces::audio::MixingBehavior::BEHAVIOR_DUCK:
89  stream << "BEHAVIOR_DUCK";
90  break;
91  }
92  stream << ", Channel:" << mediaDescription.focusChannel << ", ";
93  stream << ", TrackId:" << mediaDescription.trackId;
94  if (mediaDescription.caption.hasValue()) {
95  stream << ", CaptionData:{format:" << (mediaDescription.caption.value()).format;
96  stream << ", content:" << (mediaDescription.caption.value()).content << "}";
97  }
98  if (mediaDescription.analyzers.hasValue()) {
99  stream << ", Analyzers:{";
100  const auto analyzersCopy = mediaDescription.analyzers.value();
101  for (auto iter = analyzersCopy.begin(); iter != analyzersCopy.end(); iter++) {
102  stream << "{name:" << (*iter).name;
103  stream << ", enableState:" << (*iter).enableState << "}";
104  }
105  stream << "}";
106  }
107  stream << ", AdditionalData:{";
108  const auto additionalDataCopy = mediaDescription.additionalData;
109  for (auto iter = additionalDataCopy.begin(); iter != additionalDataCopy.end(); iter++) {
110  stream << "{" << (*iter).first;
111  stream << ":" << (*iter).second << "}";
112  }
113  stream << "}, enabled: " << (mediaDescription.enabled ? "true" : "false") << " }";
114  return stream;
115 }
