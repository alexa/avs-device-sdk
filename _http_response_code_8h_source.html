88 inline bool isStatusCodeSuccess(HTTPResponseCode code) {
89  return (HTTPResponseCode::SUCCESS_START_CODE <= code) && (code <= HTTPResponseCode::SUCCESS_END_CODE);
90 }

98 inline bool isRedirect(HTTPResponseCode code) {
99  // Not all 3xx codes are redirects and we should avoid proxy codes for security reasons. Therefore the safest way
100  // to check redirection is verifying code by code.
101  return HTTPResponseCode::REDIRECTION_MULTIPLE_CHOICES == code ||
102  HTTPResponseCode::REDIRECTION_MOVED_PERMANENTLY == code || HTTPResponseCode::REDIRECTION_FOUND == code ||
103  HTTPResponseCode::REDIRECTION_SEE_ANOTHER == code ||
104  HTTPResponseCode::REDIRECTION_TEMPORARY_REDIRECT == code ||
105  HTTPResponseCode::REDIRECTION_PERMANENT_REDIRECT == code;
106 }

115 inline HTTPResponseCode intToHTTPResponseCode(int code) {
116  switch (code) {
117  case 200:
118  return HTTPResponseCode::SUCCESS_OK;
119  case 201:
120  return HTTPResponseCode::SUCCESS_CREATED;
121  case 202:
122  return HTTPResponseCode::SUCCESS_ACCEPTED;
123  case 204:
124  return HTTPResponseCode::SUCCESS_NO_CONTENT;
125  case 206:
126  return HTTPResponseCode::SUCCESS_PARTIAL_CONTENT;
127  case 300:
128  return HTTPResponseCode::REDIRECTION_MULTIPLE_CHOICES;
129  case 301:
130  return HTTPResponseCode::REDIRECTION_MOVED_PERMANENTLY;
131  case 302:
132  return HTTPResponseCode::REDIRECTION_FOUND;
133  case 303:
134  return HTTPResponseCode::REDIRECTION_SEE_ANOTHER;
135  case 307:
136  return HTTPResponseCode::REDIRECTION_TEMPORARY_REDIRECT;
137  case 308:
138  return HTTPResponseCode::REDIRECTION_PERMANENT_REDIRECT;
139  case 400:
140  return HTTPResponseCode::CLIENT_ERROR_BAD_REQUEST;
141  case 403:
142  return HTTPResponseCode::CLIENT_ERROR_FORBIDDEN;
143  case 429:
144  return HTTPResponseCode::CLIENT_ERROR_THROTTLING_EXCEPTION;
145  case 500:
146  return HTTPResponseCode::SERVER_ERROR_INTERNAL;
147  case 503:
148  return HTTPResponseCode::SERVER_UNAVAILABLE;
149  }
150  logger::acsdkError(
151  logger::LogEntry("HttpResponseCodes", __func__).d("code", code).m("Unknown HTTP response code."));
152  return HTTPResponseCode::HTTP_RESPONSE_CODE_UNDEFINED;
153 }
