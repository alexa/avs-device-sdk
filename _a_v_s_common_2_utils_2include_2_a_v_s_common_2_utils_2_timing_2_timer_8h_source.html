231 template <typename Rep, typename Period, typename Task, typename... Args>
232 bool Timer::start(
233  const std::chrono::duration<Rep, Period>& delay,
234  const std::chrono::duration<Rep, Period>& period,
235  PeriodType periodType,
236  size_t maxCount,
237  Task task,
238  Args&&... args) {
239  if (delay < std::chrono::duration<Rep, Period>::zero()) {
240  logger::acsdkError(logger::LogEntry(TAG, "startFailed").d("reason", "negativeDelay"));
241  return false;
242  }
243  if (period < std::chrono::duration<Rep, Period>::zero()) {
244  logger::acsdkError(logger::LogEntry(TAG, "startFailed").d("reason", "negativePeriod"));
245  return false;
246  }
247 
248  // Can't start if already running.
249  if (!activate()) {
250  logger::acsdkError(logger::LogEntry(TAG, "startFailed").d("reason", "timerAlreadyActive"));
251  return false;
252  }
253 
254  // Join old timer thread (if any).
255  if (m_thread.joinable()) {
256  m_thread.join();
257  }
258 
259  // Remove arguments from the task's type by binding the arguments to the task.
260  using BoundTaskType = decltype(std::bind(std::forward<Task>(task), std::forward<Args>(args)...));
261  auto boundTask = std::make_shared<BoundTaskType>(std::bind(std::forward<Task>(task), std::forward<Args>(args)...));
262 
263  // Remove the return type from the task by wrapping it in a lambda with no return value.
264  auto translatedTask = [boundTask]() { boundTask->operator()(); };
265 
266  // Kick off the new timer thread.
267  m_thread = std::thread{
268  std::bind(&Timer::callTask<Rep, Period>, this, delay, period, periodType, maxCount, translatedTask)};
269 
270  return true;
271 }

283 template <typename Rep, typename Period, typename Task, typename... Args>
284 auto Timer::start(const std::chrono::duration<Rep, Period>& delay, Task task, Args&&... args)
285  -> std::future<decltype(task(args...))> {
286  // Can't start if already running.
287  if (!activate()) {
288  logger::acsdkError(logger::LogEntry(TAG, "startFailed").d("reason", "timerAlreadyActive"));
289  using FutureType = decltype(task(args...));
290  return std::future<FutureType>();
291  }
292 
293  // Join old timer thread (if any).
294  if (m_thread.joinable()) {
295  m_thread.join();
296  }
297 
298  // Remove arguments from the task's type by binding the arguments to the task.
299  auto boundTask = std::bind(std::forward<Task>(task), std::forward<Args>(args)...);
300 
301  /*
302  * Create a std::packaged_task with the correct return type. The decltype only returns the return value of the
303  * boundTask. The following parentheses make it a function call with the boundTask return type. The package task
304  * will then return a future of the correct type.
305  */
306  using PackagedTaskType = std::packaged_task<decltype(boundTask())()>;
307  auto packagedTask = std::make_shared<PackagedTaskType>(boundTask);
308 
309  // Remove the return type from the task by wrapping it in a lambda with no return value.
310  auto translatedTask = [packagedTask]() { packagedTask->operator()(); };
311 
312  // Kick off the new timer thread.
313  static const size_t once = 1;
314  m_thread = std::thread{
315  std::bind(&Timer::callTask<Rep, Period>, this, delay, delay, PeriodType::ABSOLUTE, once, translatedTask)};
316 
317  return packagedTask->get_future();
318 }

320 template <typename Rep, typename Period>
321 void Timer::callTask(
322  std::chrono::duration<Rep, Period> delay,
323  std::chrono::duration<Rep, Period> period,
324  PeriodType periodType,
325  size_t maxCount,
326  std::function<void()> task) {
327  // Timepoint to measure delay/period against.
328  auto now = std::chrono::steady_clock::now();
329 
330  // Flag indicating whether we've drifted off schedule.
331  bool offSchedule = false;
332 
333  for (size_t count = 0; maxCount == FOREVER || count < maxCount; ++count) {
334  auto waitTime = (0 == count) ? delay : period;
335  {
336  std::unique_lock<std::mutex> lock(m_waitMutex);
337 
338  // Wait for stop() or a delay/period to elapse.
339  if (m_waitCondition.wait_until(lock, now + waitTime, [this]() { return m_stopping; })) {
340  m_stopping = false;
341  m_running = false;
342  return;
343  }
344  }
345 
346  switch (periodType) {
347  case PeriodType::ABSOLUTE:
348  // Update our estimate of where we should be after the delay.
349  now += waitTime;
350 
351  // Run the task if we're still on schedule.
352  if (!offSchedule) {
353  task();
354  }
355 
356  // If the task runtime put us off schedule, skip the next task run.
357  if (now + period < std::chrono::steady_clock::now()) {
358  offSchedule = true;
359  } else {
360  offSchedule = false;
361  }
362  break;
363 
364  case PeriodType::RELATIVE:
365  task();
366  now = std::chrono::steady_clock::now();
367  break;
368  }
369  }
370  m_stopping = false;
371  m_running = false;
372 }
