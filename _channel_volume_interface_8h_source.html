147 inline bool ChannelVolumeInterface::adjustUnduckedVolume(int8_t delta) {
148  if ((delta > avsCommon::avs::speakerConstants::AVS_ADJUST_VOLUME_MAX) ||
149  (delta < avsCommon::avs::speakerConstants::AVS_ADJUST_VOLUME_MIN)) {
150  return false;
151  }
152  if (0 == delta) {
154  return true;
155  }
156  avsCommon::sdkInterfaces::SpeakerInterface::SpeakerSettings settings;
157  if (!getSpeakerSettings(&settings)) {
158  return false;
159  }
160  int volume = static_cast<int>(settings.volume) + static_cast<int>(delta);
161  if (volume > avsCommon::avs::speakerConstants::AVS_SET_VOLUME_MAX) {
162  return setUnduckedVolume(avsCommon::avs::speakerConstants::AVS_SET_VOLUME_MAX);
163  } else if (volume < avsCommon::avs::speakerConstants::AVS_SET_VOLUME_MIN) {
164  return setUnduckedVolume(avsCommon::avs::speakerConstants::AVS_SET_VOLUME_MIN);
165  }
166  return setUnduckedVolume(static_cast<int8_t>(volume));
167 }
