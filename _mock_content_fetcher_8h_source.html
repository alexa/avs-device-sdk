255  MockContentFetcher(const std::string& url) : m_url{url} {

258  std::unique_ptr<avsCommon::utils::HTTPContent> getContent(
259  FetchOptions fetchOption,
260  std::shared_ptr<avsCommon::avs::attachment::AttachmentWriter> writer) {
261  if (fetchOption == FetchOptions::CONTENT_TYPE) {
262  auto it1 = urlsToContentTypes.find(m_url);
263  if (it1 == urlsToContentTypes.end()) {
264  return nullptr;
265  } else {
266  std::promise<long> statusPromise;
267  auto statusFuture = statusPromise.get_future();
268  statusPromise.set_value(HTTP_REQUEST_OK);
269  std::promise<std::string> contentTypePromise;
270  auto contentTypeFuture = contentTypePromise.get_future();
271  contentTypePromise.set_value(it1->second);
272  return avsCommon::utils::memory::make_unique<avsCommon::utils::HTTPContent>(
273  avsCommon::utils::HTTPContent{std::move(statusFuture), std::move(contentTypeFuture), nullptr});
274  }
275  } else if (fetchOption == FetchOptions::ENTIRE_BODY) {
276  auto it2 = urlsToContent.find(m_url);
277  if (it2 == urlsToContent.end()) {
278  return nullptr;
279  } else {
280  static bool liveStreamPlaylistRequested = false;
281  if (m_url == TEST_HLS_LIVE_STREAM_PLAYLIST_URL) {
282  if (!liveStreamPlaylistRequested) {
283  it2->second = TEST_HLS_LIVE_STREAM_PLAYLIST_CONTENT_1;
284  liveStreamPlaylistRequested = true;
285  } else {
286  it2->second = TEST_HLS_LIVE_STREAM_PLAYLIST_CONTENT_2;
287  }
288  }
289  std::promise<long> statusPromise;
290  auto statusFuture = statusPromise.get_future();
291  statusPromise.set_value(HTTP_REQUEST_OK);
292  std::promise<std::string> contentTypePromise;
293  auto contentTypeFuture = contentTypePromise.get_future();
294  contentTypePromise.set_value("");
295  return avsCommon::utils::memory::make_unique<avsCommon::utils::HTTPContent>(
296  avsCommon::utils::HTTPContent{
297  std::move(statusFuture), std::move(contentTypeFuture), writeStringIntoAttachment(it2->second)});
298  }
299  } else {
300  return nullptr;
301  }
302  }
303 
304 private:
305  std::shared_ptr<avsCommon::avs::attachment::InProcessAttachment> writeStringIntoAttachment(
306  const std::string& string) {
307  static int id = 0;
308  std::shared_ptr<avsCommon::avs::attachment::InProcessAttachment> stream =
309  std::make_shared<avsCommon::avs::attachment::InProcessAttachment>(std::to_string(id++));
310  if (!stream) {
311  return nullptr;
312  }
313  auto writer = stream->createWriter();
314  if (!writer) {
315  return nullptr;
316  }
317  avsCommon::avs::attachment::AttachmentWriter::WriteStatus writeStatus;
318  writer->write(string.data(), string.size(), &writeStatus);
319  return stream;
320  };

323 };
