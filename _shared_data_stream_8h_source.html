312 template <typename T>
313 size_t SharedDataStream<T>::calculateBufferSize(size_t nWords, size_t wordSize, size_t maxReaders) {
314  if (0 == nWords) {
315  logger::acsdkError(logger::LogEntry(TAG, "calculateBufferSizeFailed").d("reason", "numWordsZero"));
316  return 0;
317  } else if (0 == wordSize) {
318  logger::acsdkError(logger::LogEntry(TAG, "calculateBufferSizeFailed").d("reason", "wordSizeZero"));
319  return 0;
320  }
321  size_t overhead = BufferLayout::calculateDataOffset(wordSize, maxReaders);
322  size_t dataSize = nWords * wordSize;
323  return overhead + dataSize;
324 }

336  logger::acsdkError(logger::LogEntry(TAG, "createFailed").d("reason", "nullBuffer"));
337  return nullptr;
338  } else if (expectedSize > buffer->size()) {
339  logger::acsdkError(logger::LogEntry(TAG, "createFailed")
340  .d("reason", "bufferSizeTooSmall")
341  .d("bufferSize", buffer->size())
342  .d("expectedSize", expectedSize));
343  return nullptr;
344  }

346  std::unique_ptr<SharedDataStream<T>> sds(new SharedDataStream<T>(buffer));
347  if (!sds->m_bufferLayout->init(wordSize, maxReaders)) {
348  // Logged in init().
349  return nullptr;
350  }

381 std::unique_ptr<typename SharedDataStream<T>::Writer>
382 SharedDataStream<T>::createWriter(typename Writer::Policy policy, bool forceReplacement) {
383  auto header = m_bufferLayout->getHeader();
384  std::lock_guard<Mutex> lock(header->writerEnableMutex);
385  if (header->isWriterEnabled && !forceReplacement) {
386  logger::acsdkError(logger::LogEntry(TAG, "createWriterFailed")
387  .d("reason", "existingWriterAttached")
388  .d("forceReplacement", "false"));
389  return nullptr;
390  } else {
391  return std::unique_ptr<Writer>(new Writer(policy, m_bufferLayout));
392  }
393 }

395 template <typename T>
396 std::unique_ptr<typename SharedDataStream<T>::Reader>
397 SharedDataStream<T>::createReader(typename Reader::Policy policy, bool startWithNewData) {
398  std::unique_lock<Mutex> lock(m_bufferLayout->getHeader()->readerEnableMutex);
399  for (size_t id = 0; id < m_bufferLayout->getHeader()->maxReaders; ++id) {
400  if (!m_bufferLayout->isReaderEnabled(id)) {
401  return createReaderLocked(id, policy, startWithNewData, false, &lock);
402  }
403  }
404  logger::acsdkError(logger::LogEntry(TAG, "createWriterFailed").d("reason", "noAvailableReaders"));
405  return nullptr;
406 }

424 template <typename T>
425 std::unique_ptr<typename SharedDataStream<T>::Reader>
426 SharedDataStream<T>::createReaderLocked(
427  size_t id,
428  typename Reader::Policy policy,
429  bool startWithNewData,
430  bool forceReplacement,
431  std::unique_lock<Mutex> * lock) {
432  if (m_bufferLayout->isReaderEnabled(id) && !forceReplacement) {
433  logger::acsdkError(logger::LogEntry(TAG, "createReaderLockedFailed")
434  .d("reason", "readerAlreadyAttached")
435  .d("readerId", id)
436  .d("forceReplacement", "false"));
437  return nullptr;
438  } else {
439  // Note: Reader constructor does not call updateUnconsumedCursor() automatically, because we may be seeking to
440  // a blocked writer's cursor below (if !startWithNewData), and we don't want the writer to start moving before
441  // we seek.
442  auto reader = std::unique_ptr<Reader>(new Reader(policy, m_bufferLayout, id));
443  lock->unlock();
444 
445  if (startWithNewData) {
446  // We're not moving the cursor again, so call updateUnconsumedCursor() now.
447  m_bufferLayout->updateOldestUnconsumedCursor();
448  } else {
449  Index offset = m_bufferLayout->getDataSize();
450  if (m_bufferLayout->getHeader()->writeStartCursor < offset) {
451  offset = m_bufferLayout->getHeader()->writeStartCursor;
452  }
453  // Note: seek() will call updateUnconsumedCursor().
454  if (!reader->seek(offset, Reader::Reference::BEFORE_WRITER)) {
455  // Logged in seek().
456  return nullptr;
457  }
458  }
459  return reader;
460  }
